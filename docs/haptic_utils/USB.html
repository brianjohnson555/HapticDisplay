 97def make_packet(duties: np.ndarray, periods: np.ndarray):
 98    """Converts the duty cycle and period arrays into a USB-transferable packet.
 99    
100    **Parameters** :
101
102    >>>**duties** : np.ndarray of shape (10,) containing the duty cycle [%] of each MINI switch
103    
104    >>>**periods** : np.ndarray of shape (10,) containing the total period [ms] of each MINI switch
105    
106    **Returns** :
107    >>>**packet** : combined byte string containing both period and total time [ms]"""
108
109    duties_abs = np.int32(np.floor(np.multiply(periods,duties))) # convert from % to msec
110    packetlist = []
111    
112    # encode start of period array
113    packetlist.append(('P').encode()) 
114    for duty in duties_abs:
115        packetlist.append((duty.item()).to_bytes(2, byteorder='little')) # convert to 16bit
116
117    # encode start of total array
118    packetlist.append(('T').encode()) 
119    for period in periods:
120        packetlist.append((period.item()).to_bytes(2, byteorder='little')) # convert to 16bit
121    packet = b''.join(packetlist) # combine packetlist as bytes
122
123    return packet

125def make_packet_list(duty_array: np.ndarray = np.zeros((30,),dtype=int), period_array: np.ndarray = np.zeros((30,),dtype=int)):
126    """Converts the duty cycle and period arrays into a list of USB-transferable packets.
127    This assumes mapping from (30,) array to 3 serial ports as explained in README.md of this repository.
128    
129    **Parameters** :
130
131    >>>**duties** : np.ndarray of shape (30,) containing the duty cycle [%] of each MINI switch
132    
133    >>>**periods** : np.ndarray of shape (30,) containing the total period [ms] of each MINI switch
134    
135    **Returns** :
136    >>>**packet_list** : list of combined byte strings containing both period and total time [ms]
137    """
138    
139    packet_list = []
140    #iterate for each serial port:
141    for ii in [0, 10, 20]:
142        # Parse duty cycles and periods from input:
143        duties = duty_array[ii:ii+10]
144        periods = period_array[ii:ii+10]
145
146        duties_abs = np.int32(np.floor(np.multiply(periods,duties))) # convert from % to msec
147        packetlist = []
148        packetlist.append(('P').encode()) # encode start of period array
149        for duty in duties_abs:
150            packetlist.append((duty.item()).to_bytes(2, byteorder='little')) # convert to 16bit
151
152        packetlist.append(('T').encode()) # encode start of total array
153        for period in periods:
154            packetlist.append((period.item()).to_bytes(2, byteorder='little')) # convert to 16bit
155        packet = b''.join(packetlist) # combine packetlist as bytes
156
157        # append to output:
158        packet_list.append(packet)
159
160    return packet_list

162def make_packet_sequence(duty_array_list:list, period_array_list:list):
163    """Converts lists of duty cycle and period information into a list of USB packets.
164    Intention is to pre-process all packets before running real-time codes.
165    
166    **Parameters** :
167
168    >>>**duty_array_list** : list of duty cycle arrays. Each list element is a flat np.ndarray of shape (30,)
169    
170    >>>**period_array_list** : list of period arrays. Each list element is a flat np.ndarray of shape (30,)
171    
172    **Returns** :
173    
174    >>>**packet_sequence** : list of packet_lists. Each element of *packet_sequence* is a list with
175     three elements *[packet_A, packet_B, packet_C]* with A/B/C corresponding to USB serial output."""
176    packet_sequence = []
177
178    for duty_array, period_array in zip(duty_array_list, period_array_list):
179        # convert to packets:
180        packet_list = make_packet_list(duty_array, period_array)
181        # append:
182        packet_sequence.append(packet_list)
183
184    return packet_sequence
